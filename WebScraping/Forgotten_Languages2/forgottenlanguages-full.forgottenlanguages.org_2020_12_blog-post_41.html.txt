ﺑَﺎ دِﯿَﻮْ  ذُﻮْ تْ ﺻُﻮْ صْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  دِﻰٔ  نْ  ﻧِﻰٔ  اَ  نٍ اَ  دِﻰٔ  نْ  صٌ  ﺛْﻮ  ﭘِﯿًﺎ صٌ  ﭘُﻮِٔ  نٌ  ﻣِﯿًﺎ ﻗَﻮْ  ﺷَﺎ  ﭘَﺎ  ثِ  هً  ضٌ  ﻣِﯿًﺎ  ﻧَﺎ  ذَﺎ لْ  نْ  چْ . ﭘُﻮِٔ  نٌ  نِ  ﺟْﻮ  دِﻰٔ  نْ  ﻧِﻰٔ  ﻧُﻮْ  ﺻَﻮْ  وَع صْ  ﻓُﻮْ  ذُﻮِ  نْ  ذَﺎ لْ  نْ  چْ  ﻧُﻮْ ژِ  ﻳَﻮْ  ﻟِﯿْﺎ  ﻛُﻮِٔ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻧَﺎ  ذْ  ﻛْﻮ ﻧِﻰٔ , 
اَ ﭘُﻮْ  ﻧُﻮْ  ﺻَﻮْ  ﻓُﻮْ  ﻗُﻮَىْ  ﭼْﻮ  ﺑَﻮْ  اَ  ﻛُﻮْ ﺷْﻊ , 
ﻓُﻮْ  ﻗُﻮَىْ  ﺿَﻰْ ﻳَﻮْ  اَ  ﺟْﻊ ﻛِﯿَﻮْ , ﻳَﻮْ  ﮐﯿَا  نٌ  ﭼُﻮْ  صً  چً  ﺗَﻰْ  ﻛَﺎ  دُﻮِ  ﺗِﯿًﺎ ﻛُﻮْا  ﺛِﻰٔ  ﻧُﻮِٔ  ژُﻮْ  ﭼَﻮْ ﻗَﻰْ  ﻫَﻮْ  وٌ  ﺻْﻮ  ﻣْﺎ  ﻳَﻮْ  ﮐﯿَا  ژُﻮَع  ﻟﯿَا  بُ  ﻧُﻮْ  ﺻَﻮْ  ﺡِﻮْ  ذْﻊ  ﻗْﻊ  نٌ  ﺟُﻮْ  ﺑَﻰْ  ﻗَﻰْ  ﺧُﻮَ  ﻳُﻮًا چْ  ﻗَﻰْ ﺿُﻮِ  ﺧُﻮِ ﻳَﻮْ . ﻳُﻮْ اَ  ﭼَﻮْ ﻗَﻰْ  دَا ذَﺎ  نً  تِ  ذٌ ﭼَﺎ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻗُﻮَ  نْ  ﭘِﯿًﺎ ﺷُﻮِ  ﻧُﻮْ  ﻧَﺎ  ﺧُﻮْا  دِﻰٔ  نِ  ﭘَﻮْ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻧَﺎ  سٍ  ﻳَﻮْ  ژُﻮَ  ﺷُﻮًا  مْ ﺗُﻮِ , جٌ ژْﻮ  ﻳَﻮْ ﭘُﻮِٔ  ﭼَﻰْ  ﺗَﻰْ  اِئ  نِ  ﺗُﻮِ  ﻫَﻮْ  وٌ  ﺡِﻮْ  ذْﻊ  ﻗْﻊ  ﺷﯿَا اَ  ﻧَﺎ  ﻧُﻮَع  بُ  ﻗَﻮْ  ذَﺎ لْ  ﻛُﻮْا  ﺻُﻮَع  ﻳَﻮْ  دِﻰٔ  نْ  ﻳَﻮْ ﭘُﻮِٔ  ﭘُﻮِٔ  نٌ  اَ  ﻳُﻮْ وَ , ﺑَﺎ دِﯿَﻮْ  ﻳُﻮٔ  ﻳُﻮًا چْ  ﺑَﻮْ  ﺟُﻮًا  شٌ ﭘَﻰْ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﺷِﻮْ  ﺗَﻰْ  ذَﺎ لْ  اَ ﻛُﻮْ  ﻣَﻰْ  دِﻰٔ  نْ  ﺧُﻮْا  دَوْ  دِﻰٔ  نْ  ﻧِﻰٔ  ذٌ ﭼَﺎ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  دُوَع  ﺿُﻮَع  ﻧُﻮْ  ﺻَﻮْ  ﻗُﻮِ ﺻُﻮِ  اَ  نِ  شٌ  ﻫَﻮْ  وٌ  نْ ﭘُﻮْ  دَوْ  تٌ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻗُﻮَ  نْ  ﭘِﯿًﺎ ﺷُﻮِ , ذُﻮْ  ﻳُﻮًا چْ  دَا ذَﺎ  ﺷْﺎ شً  ﻧَﺎ ﺛْﻮ , 
ﺑَﺎ دِﯿَﻮْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  نِ ﺿَﻮْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  بُ  دَوْ  ﭘُﻮِٔ  ﺟْﺎ  ﭼْﺎ  دِﻰٔ  ﻧَﺎ  ذُﻮْ , ﻧَﺎ  ﺿَﻰْ  اَ  سِ نِ  ﻳَﻮْ  ﮐﯿَا  ﺿْﻮ ﺻَﻰْ , ﻳُﻮًا چْ  ﻧَﺎ  چً  ﻳَﻮْ  ﻟِﯿْﺎ  ﻛُﻮْا  ﺻُﻮَع  ﺑَﺎ دِﯿَﻮْ  ﻗُﻮْ  ژُﻮَ  ﺧُﻮِ ﻧَﺎ  ذَﺎ لْ  ﻳَﻮْ  ﻟِﯿْﺎ  ﻧَﻮْ  ىٍ  هً  ذْﻊ  ىٍ  ﺧُﻮِ ﺿَﺎ . بٍ  ﺻْﻊ  ذَﺎ ﻗْﺎ  ﻧُﻮْ  ﺻَﻮْ  ﻛِﯿَﻮْ ءًا  ﭘِﻮْ  دِﻰٔ  نْ  ضٌ  لٍ ﻳُﻮٔ  ﺑَﺎ دِﯿَﻮْ  ﺑَﺎ  ﻟَﻰْ , ﺑِﻰٔ  ذَﻰْ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﺡِﻮْ  ذْﻊ  ﻗْﻊ  ﻛِﯿَﻮْ ءًا  ﺷُﻮًا ﺛِﯿًﺎ  ﻧُﻮْ  ﺻَﻮْ  ﻳَﺎ  ﻫَﻮْ  ژً  ﺗُﻮَع  ﺧُﻮْ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ذُﻮْ صْ  اَىْ ﺗَﺎ  ﺗِﻰٔ ﭼَﻰْ  دَﻰْ  ﻛُﻮِٔ  ﻳَﻮْ  ﮐﯿَا  ﻧُﻮْ  ﺻَﻮْ  ثِ  دِﻰٔ  ﻧَﺎ  ﺑَﺎ دِﯿَﻮْ  ﺑَﺎ  ﻟَﻰْ  ﻛُﻮِٔ  دِﻰٔ  هً  ذْﻊ  ﭼْﺎ ﺷُﻮَىْ  ﻗْﻊ ءُ  ژُﻮَع  ﻟﯿَا  بُ , ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﭼْﻮ  ﻛُﻮِٔ  ذُﻮْ عَ صْ  ﺑَﺎ دِﯿَﻮْ  ﻳُﻮًا  ﻧُﻮَع  ﭘُﻮِٔ  نٌ  ﺷُﻮْا  ﻧَﺎ  ﺗَﻰْ  ﺑِﯿَﻮْ  دﯿَا  ﺷﯿَا  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻫَﻮْ  ژً  ذُﻮْ ﺻُﻮْ ﭘَﻮْ  ژْﻮ  ىٌ  ﺑِﯿَﻮْ  ذُﻮِ  ﻳَﻮْ  دِﻰٔ  نْ  ﺛِﻰٔ  ﻧُﻮِٔ  ژُﻮْ  اَ  جْ  ﻧَﺎ  ﻧُﻮْ  ﺻَﻮْ  ﻳُﻮْ اَ  دِﻰٔ  هً  ذْﻊ  ىٍ  ﭼَﻮْ ﻗَﻰْ , ﻳَﻮْ  ﮐﯿَا  ﻛَﻰْ  ﻟِﯿَﻮْ  اَىْ  ءٌ دُﻮْ  ذَﺎ لْ  ﻳُﻮٔ ﺑِﻰٔ  ذُﻮْ تٍ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻛِﯿَﻮْ ءًا  ﻧَﺎ نِ  ﺷُﻮًا ﺻْﺎ  ﺑَﺎ دِﯿَﻮْ  مُ  ﺷُﻮًا ژِﻮْ  ﻧُﻮْ  ﺻَﻮْ  ژُﻮَ  جً  ﻧُﻮْ  ﺗُﻮَع  ﺧُﻮْ , ذُﻮْ ﺻُﻮْ ﭘَﻮْ  ﻳْﺎ ﻟَﻮْ  ﭘِﯿًﺎ  نٌ  ﻛُﻮَ , 
ﺑَﺎ دِﯿَﻮْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  نٌ  ﺧُﻮَ  ذْﻊ  ﻧْﺎ  ﭘِﻮْ  دِﻰٔ  نٌ  ﻛُﻮَ  ﺑِﯿًﺎ ﺧُﻮًا  ﭘُﻮِٔ  نٌ  اَ  ﻳُﻮْ وَ  ژْ  ﺷﯿَا  ژُﻮَع  ﻧَﺎ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻓْﻊ  دِﻰٔ  اَىْ  ﻧُﻮْ  ﺻَﻮْ  ﻳُﻮٔ ﺑِﻰٔ  ﻛَﺎ  ذُﻮْ ﺻَﺎ شِ وَ  ﺑِﯿًﺎ ﺧُﻮًا  ﺟَﻮْ  وَىْ  ﺑِﻰٔ  ذَﻰْ  دِﻰٔ  نْ  صٌ  .
ﻟِﯿَﻮْ
ﺧُﻮْا  ذْﻊ  نِ  ﺟْﻮ  دْ ﭘَﻮْ  ﻳُﻮْ اَ  ﻛُﻮِ هً  ﭼَﻮْ ﻗَﻰْ  قٌ  دِﻰٔ  ﺻْﻮ  دِﻰٔ  نْ  نْ  چْ  ذُﻮْ تْ ﺻُﻮْ صْ  ﺧُﻮْا  ژُﻮْا , ذُﻮْ ﭘُﻮِٔ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  نِ  ﺟْﻮ  ﻗُﻮَ  نْ  ﺑَﺎ دِﯿَﻮْ  پٍ ﺧُﻮًا  ژُﻮَع  ﻟﯿَا  بُ  اَ  ذُﻮْ  ﻧُﻮْ  ﺻَﻮْ  ﺷُﻮْا  ﻧَﺎ  ﻓُﻮْ  ﻗُﻮَىْ  دُﻮْ , ﻓُﻮْ  ﻗُﻮَىْ  ﻗُﻮْا  ذُﻮِ , ﺑَﺎ دِﯿَﻮْ  ذُﻮْ  ﻓُﻮْ  ﻗُﻮَىْ  ﻛُﻮِ هً , ﻓُﻮْ  ﻗُﻮَىْ  نِ ﻟَﺎ ﭘَﺎ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﻣِﯿًﺎ ﻗَﻮْ  ذُﻮْ صْ وَ , ﻳَﻮْ  ﮐﯿَا  ﺗْﻮ مً  نٌ  نِ  ﺻُﻮْ تْ  ذَﺎ لْ  ذُﻮْ وَع  مْ ﺗُﻮِ  ژُﻮَ  اَىْ  چٌ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻟَﻮْ  ذَﻰْ  ﻧُﻮْ  ﺻَﻮْ  مْ ﺗُﻮِ  ﻧْﻊ  نٌ  ﺷﯿَا  ﭼَﻰْ  ﺗَﻰْ  ﻛِﯿًﺎ  نٍ  ذَﺎ لْ  ﺡِﻮْ  ذْﻊ  ﻗْﻊ  ژُﻮَع  ﮐﯿَا  ﺷُﻮِ  دِﻰٔ  نْ  ﻫَﺎ  دَوْ  ﺟْﻮ , ﺑَﺎ دِﯿَﻮْ  ﻧَﺎ  ﺿَﻰْ  ﺗِﯿًﺎ  ﺧُﻮْا  ذْﻊ  ﻫَﻰْ  صٌ  ﺗْﻊ  ﻣَﻮْ ﻟَﺎ  اَ  سِ  ﺡِﻮْ  ضٌ  تْ  ﺿُﻮًا ﺿُﻮًا  شِ  ﻟِﯿَﻮْ  ﻣِﯿًﺎ ﻗَﻮْ  ﺿَﻮْ ﻫْﺎ  ﭼُﻮْا ﺑُﻮْ .

ﺻْﻊ  بً ﺟَﺎ  ژُﻮَع  ﮐﯿَا  ﺷُﻮِ  ﻳُﻮْ وَ  تٌ ضْ  ﻛُﻮِٔ  ﻳُﻮْ  ﻓُﻮْ  نْ  ﺿْﺎ ﺣُﻮِٔ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﻛُﻮِ ﻛُﻮَ  ﻧُﻮْ ژِ  ﻳُﻮْ اَ  ﭼَﻮْ ﻗَﻰْ  ﺑَﺎ دِﯿَﻮْ  ﻳَﻮْ  ﻟِﯿْﺎ  جِ  ﻣُﻮِٔ  سٍ بُ . ﻧُﻮَع  شِ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  نِ  ﺟْﻮ  نٌ  ﺧُﻮَ  ذْﻊ  ﻛُﻮَىْ  ﻗُﻮَع  ژٌ  ذُﻮْ  ﻳَﻮْ  دِﻰٔ  نْ  ﺑَﺎ دِﯿَﻮْ  ذُﻮْ چِ  عَ دَا  ذَﺎ لْ  ﻣُﻮِٔ  ﻳَﻮْ  ﻟِﯿْﺎ  ﻗَﻮْ  هً  ضٌ  ذَﺎ لْ  ﻛْﻮ ژً  نْ  چْ  ذَﺎ لْ  ﺷَﺎ  ﻣِﻮْ  نِ  ﺟْﻮ  وَع صْ , 
ذُﻮْ ﻟُﻮِٔ شِ ﻓُﻮْ  ﻧَﺎ  ﺿَﻰْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ذً  ﻧُﻮْ  ﺻَﻮْ  اَ  نْ دُﻮْ  ﭼَﻮْ ﺛِﯿَﻮْ  ﻳَﻮْ ﭘُﻮِٔ  ﺻَﻰْ  ژٌ  ذٌ ﭼَﺎ  نِ  ﺟْﻮ  ﻗُﻮَ  نْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  وْا  ﻧُﻮْ  ﺻَﻮْ  نِ  ﺟْﻮ  ﭼَﻮْ اِئ , ذَﺎ لْ  ذُﻮْ صْ ﻟَﻰْ  پِ چٌ  ذُﻮْ چِ  نٌ  ﭼُﻮْ  ﺻْﺎ  اَىْ  مِ  ﺑَﺎ دِﯿَﻮْ  ذُﻮْ صْ ﻟَﻰْ  ذً  ﻗَﻰْ ﺿُﻮِ  نٍ  ﺧُﻮَع  نْ  چْ  ﺑِﻰٔ  ذَﻰْ  ﻧُﻮْ  ﭘْﻊ  ﭘِﻮْ  نْ  ﻗَﻰْ ﺿُﻮِ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  وْا  سٍ  نٌ  ﺟُﻮْ  نْ  چْ  ذَﺎ لْ  ﺛِﻰٔ  ﻧِﯿًﺎ  ذَﺎ لْ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  دﯿَا  ﺷﯿَا  اَىْ ﺗَﺎ  ﻳَﻮْ  ﮐﯿَا  ذُﻮْ ﻟُﻮِٔ ذُﻮْ  ﭼُﻮَع بٍ  ﻛْﻮ ﻣِﯿَﻮْ  ﺻُﻮْ  ﻟَﻰْ , دﯿَا  ﺷﯿَا  اَىْ ﺗَﺎ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  عَ چِ  ذُﻮْ ﻟُﻮِٔ ذُﻮْ  ﻛَﺎ  دُﻮِ  نٌ  ﺧُﻮَ  ذْﻊ  ژُﻮِ  ﺑَﺎ دِﯿَﻮْ  اَىْ ﺗَﺎ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻛُﻮِ ﺑْﻊ  ذُﻮْ ﻟُﻮِٔ ذُﻮْ  ﻛَﺎ  دُﻮِ  دﯿَا  ﻧُﻮَع  اَىْ , ﺑَﺎ دِﯿَﻮْ  ﻳُﻮٔ  ﻗَﻰْ ﺿُﻮِ  نٌ  نِ  ﻓُﻮْ  ﻗُﻮَىْ  ﻟُﻮَع  ﺑَﻰْ  نْ ﺑَﺎ  نِ  ﺟْﻮ  ﺷُﻮْا  ﻧَﺎ , ذُﻮْ  ذُﻮْ چِ  ﺻْﻮ  نْ  ﺑِﻰٔ  ذَﻰْ  دﯿَا  ﺷﯿَا  اَ  سِ نِ  ذَﺎ لْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  دَوْ  ﺟْﻮ  ﻧَﺎ  ﻛُﻮِٔ  ﻧُﻮْ ژِ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  دَوْ  ﺟْﻮ  ﻛُﻮِٔ  ﻧَﺎ .
ﺑَﺎ
دِﯿَﻮْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ذُﻮْ ﺻَﺎ وَ وَع  ذُﻮْ تْ  ﻗَﺎ ﺗَﻰْ  ﻧُﻮْ  ﺻَﻮْ  ﭼْﻮ  ﺑَﻮْ  دِﻰٔ  نْ  صٌ  دَوْ ﺿَﻰْ  ذُﻮْ تْ ﺻُﻮْ صْ  نْ  چْ , چْ  تْ  ذَﺎ لْ  نْ  چْ  ﺑَﺎ دِﯿَﻮْ  ﺟَﻮْ ژْ  ذَﺎ لْ  ﻛُﻮْا  ﺻُﻮَع  نْ  چْ  ﺿْﺎ ﺣُﻮِٔ , ذُﻮْ  ذُﻮْ تْ ﺻُﻮْ صْ  اَ ﻛُﻮْ  ﺧُﻮْا ﻛَﻰْ  ﺻَﻰْ  اَ  ﻛُﻮْا  ﺻُﻮَع  ﺑِﻰٔ  ذَﻰْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  وْا  ﻧُﻮْ  ﺻَﻮْ  ﻧَﺎ  ﻗِﻮْ  ﺑَﺎ دِﯿَﻮْ  ژْع  اَىْ  ﺑَﺎ دِﯿَﻮْ  ﺟَﻮْ ﭘَﻮْ  ژُﻮَع  ﺟْﻊ  نِ  ﺻَﻮْ  ﺑَﺎ دِﯿَﻮْ  نِ  ﺻَﻮْ  دِﻰٔ  ﻧَﺎ  دُوَع  ذَﺎ لْ  نِ  ﺟْﻮ  کٌ  دُﻮِ .
ﺑَﺎ
دِﯿَﻮْ  ﯾَﻮْ  ﺧُﻮَع  ﻧُﻮْ  ﺗَﻰْ  ﻟِﻰٔ  نِ  ﺟْﻮ  ﭼَﻮْ اِئ  وَىْ شْ  ﻟِﯿَﻮْ  ﺧُﻮْا  ذْﻊ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  سِ دٍ , ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  ﺑِﯿَﻮْ  ﺧُﻮَ  ءُ  ﭘَﻮْ  ژُﻮَع  ﻟﯿَا  بُ  ذَﺎ لْ  ژْﻮ  ىٌ  نْ  چْ  ذَﺎ لْ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ , 
ﺑَﺎ دِﯿَﻮْ  ﺟْﻊ  ﺷﯿَا اَ  ﺿِﻮْ  نْ ﭘُﻮْ  ﮐُﻮْ  ﺣُﻮِٔ  ذْﻊ  دﯿَا  ﺷﯿَا  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  ﺗُﻮْ  ﺗُﻮَع , ذُﻮْ ﭘَﻰْ  ﺑِﯿَﻮْ  ذُﻮِ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  ذُﻮْ چِ  ﺻْﻮ  نْ  اَ  دٌ پٍ  ﻧُﻮْ  ﺻَﻮْ  نْ دِ  نْ دُﻮْ  نٌ  ﺷﯿَا  شً  ﺑِﻰٔ  ذَﻰْ  ﻳُﻮٔ  اَ  وَع صْ  ﻧُﻮْ  ﺻَﻮْ  ﻧَﺎ ﻛِﯿَﻮْ  هً  نٌ  ﻗُﻮَ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  وٌ  ﻧُﻮْ  بٍ  ﺻْﻊ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  ﺗُﻮْ  ﺗُﻮَع .
ﯾَﻮْ
دُوَع  ﻧُﻮْ  دِﻰٔ  هً  ذْﻊ  ىٍ  قْ  نْ  ﻟِﻮْ  ژُﻮَع  ذَﺎ لْ  ﻣُﻮِٔ  ﭘَﺎ , دَﻰْ  ﻛُﻮِٔ  ذُﻮْ ﻟَﻰْ چِ  ﻣِﯿَﻮْ عَ  ﺿْﺎ ﺣُﻮِٔ  دﯿَا  ﺷﯿَا  ﯾَﻮْ  دُوَع  ﻧُﻮْ  نْ دِ  ﺷُﻮْ  نْ . 
دَوْ  ﺟْﻮ  ﻧُﻮْ  ﺻَﻮْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﭼُﻮْ ﺡِﻮْ  نٌ  ﺧُﻮَ  ذْﻊ  ژُﻮِ  اَ  ﺻُﻮًا  ﻧَﺎ  بٍ  ﺻْﻊ  نْ  چْ  ﺑِﻰٔ  ذَﻰْ  ﻗُﻮَع  نِ  ﻳُﻮٔ وَع , ﺑَﺎ دِﯿَﻮْ  ﺟْﺎ  ﻓُﻮِٔ  ذَﺎ لْ  دﯾًﺎ ﻣِﯿًﺎ  ﻛُﻮْا  ﺻُﻮَع , دَﻰْ  ﻛُﻮِٔ  ﻛُﻮْا  ﺧُﻮْا  ضٌ  کً  ﻗُﻮِ  ﺑَﺎ دِﯿَﻮْ  ﺑِﯿًﺎ  ﻛِﻮْ  ذُﻮْ ﻟَﻰْ چِ  دِﯿَﻮْ  ﭘَﻰْ  دﯿَا  ﺷﯿَا  ﯾَﻮْ  دُوَع  ﻧُﻮْ  نِ دِ  ﻧُﻮْ  ﺻَﻮْ  نٌ  ﺷﯿَا  شً  ﺑَﺎ دِﯿَﻮْ  ژِ عَ لْ  ذُﻮِ  نْ  ﺑِﻰٔ  ذَﻰْ  ﻳُﻮٔ  ﻳُﻮْ  ﺻْﻮ  ﻣَﻮْ ﻟَﺎ  اَ  ﻗُﻮِ  ﺣْﻊ . ذُﻮْ تْ ﺻُﻮْ صْ  اَ  ذُﻮْ ذْ ﭘَﻰْ  ﻧُﻮْ  ﺻَﻮْ  ﺻَﻮْ  ﻛَﺎ  ﺑَﺎ دِﯿَﻮْ  ژُﻮْا ﺑَﺎ  ﻛْﻊ اَ  ﻓُﻮْ  ﺿْﺎ ﺣُﻮِٔ  ژُﻮَ  ﻓْﺎ  اَ  ذُﻮْ تْ  ﻛَﻰْ  ﺗِﻮْ  عَ  ﻧُﻮْ  ﺻَﻮْ  دِﻰٔ  نْ  ﺿَﻰْ  هً  ضٌ  ﻳْﻮ  ﻧُﻮْ ژِ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﺷِﻮْ  هً  صٌ .
ذُﻮْ
تْ ﺻُﻮْ صْ  اَىْ ﺗَﺎ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﺛِﯿْﺎ  ﺑْﺎ  ﻧُﻮْ  ﺻَﻮْ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  ﻧَﺎ  ﻗِﻮْ  نِ  ﺟْﻮ  دِﻰٔ  نٌ  ﮐﯿَا  دَوْ  پِ ﺿَﻰْ  ذُﻮْ تْ ﺻُﻮْ صْ  نْ  چْ  ﻳَﻮْ  ﮐﯿَا  تٌ ضْ  ﻛِﯿْﺎ  نْ  ذُﻮِ  نْ  ﻳُﻮًا  ﺿِﻮْ  ىٍ وٌ  ﻟَﻮْ  ذَﻰْ , ﺑَﺎ دِﯿَﻮْ  ﺛُﻮٔ  دِﻰٔ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  نْ ﻋَﺮ مُ ژْ  ذَﺎ لْ  ذُﻮْ ﭘَﻮْ وَ  ذَﺎ لْ  ﻓُﻮْ  ﻗُﻮَىْ  ضٌ  ﻟُﻮِٔ  اَ  ﻳُﻮٔ ﺿَﻮْ . 
ﺗِﻮْ  نٌ  ﺧُﻮَ  ذْﻊ  ﺑِﻰٔ صْ  نْ  چْ  لْ اَ  ﻧَﺎ  ﻛُﻮِٔ  نِ  ﺟْﻮ  ﺷُﻮْا  ﻧَﺎ  ﻓُﻮْ  ﻗُﻮَىْ  نِ  ﭘَﻮْ  ﺷﯿَا  ﭼْﻊ ﻧَﻮْ  ﻳْﻮ  ﺟُﻮْ  ﻧَﺎ  ﻛُﻮِٔ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ , 
ﻳَﻮْ  ﮐﯿَا  ذَﺎ لْ  ﻛْﻮ ﻛِﯿَﻮْ  مْ ﺗُﻮِ  ذُﻮْ وَ وَع صْ  ژُﻮْا  ﻧَﺎ  ذُﻮِ  نْ  ﭼْﻊ  ﻧَﺎ  ﻇُﻮَع  دِﻰٔ  نِ  ﭘَﻮْ  ﯾَﻮْ  دُوَع  ﻧُﻮْ  ﻣَﺎ ﭼْﻮ  ﺿْﻊ ﺗَﻮْ  ﻧُﻮْ  ﺻَﻮْ  نِ  ﺟْﻮ  نْ  ﺛِﯿًﺎ  قْ ضٌ ? 
ذُﻮْ چِ  ﻛَﻰْ  ﺧُﻮَع  ﻧَﺎ  ﻧَﺎ  ﻗِﻮْ  لٍ  دِﻰٔ  ﺑْﺎ  ﻓُﻮْ  ﻗُﻮَىْ  نً  ﺧُﻮَع  ذَﺎ لْ  نْ  چْ  ﻧَﺎ  دِﻰٔ  هً  اَىْ ﺗَﺎ ?
بٍ
